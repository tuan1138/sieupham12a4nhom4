        // --- 1. HIỆU ỨNG HẠT (TIM, BONG BÓNG, PHÁO HOA) ---
        const canvas = document.getElementById('particleCanvas');
        const ctx = canvas.getContext('2d');
        let particles = [];
        const types = ['❤️', '🎈', '✨'];

        function resize() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; }
        window.addEventListener('resize', resize);
        resize();

        class Particle {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = canvas.height + 100;
                this.size = Math.random() * 20 + 10;
                this.speed = Math.random() * 2 + 1;
                this.type = types[Math.floor(Math.random() * types.length)];
                this.opacity = Math.random() * 0.5 + 0.2;
                this.angle = Math.random() * Math.PI * 2;
            }
            draw() {
                ctx.globalAlpha = this.opacity;
                ctx.font = `${this.size}px serif`;
                ctx.fillText(this.type, this.x, this.y);
            }
            update() {
                this.y -= this.speed;
                this.x += Math.sin(this.angle) * 0.5;
                if (this.y < -50) {
                    this.y = canvas.height + 100;
                    this.x = Math.random() * canvas.width;
                }
            }
        }

        for (let i = 0; i < 30; i++) particles.push(new Particle());
        function animate() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particles.forEach(p => { p.update(); p.draw(); });
            requestAnimationFrame(animate);
        }
        animate();

        // --- 2. XỬ LÝ ẢNH & VIDEO ---
        const listLop = [2, 4, 5, 6, 11, 12]; // Đã lọc bỏ các số bạn yêu cầu
        const list2011 = [1, 2, 3, 4, 5, 6, 8, 10, 11]; // Bỏ 7, 9
        const listDL = Array.from({length: 18}, (_, i) => i + 1);
        const listVD = Array.from({length: 6}, (_, i) => i + 1);

        // Chèn ảnh vào các section
        const renderImgs = (ids, prefix, ext, target) => {
            const container = document.getElementById(target);
            if (!container) return;
            ids.forEach(id => {
                const url = `${prefix}${id}.${ext}`;
                const html = target.includes('slides') 
                    ? `<div class="swiper-slide"><img src="${url}"></div>`
                    : `<div class="reveal rounded-xl overflow-hidden shadow-lg"><img src="${url}" class="w-full hover:scale-110 transition duration-500"></div>`;
                container.innerHTML += html;
            });
        }

        renderImgs(listLop, 'lop', 'jpg', 'home-lop-slides');
        renderImgs(list2011, '20.11.', 'jpg', 'home-2011-slides');
        renderImgs(listDL, 'dl', 'jpg', 'album-dl-grid');
        
        // Ký ức 12 (Tổng hợp)
        const kyucContainer = document.getElementById('album-kyuc-masonry');
        const allPhotos = [
            ...listLop.map(i => `lop${i}.jpg`),
            ...list2011.map(i => `20.11.${i}.jpg`),
            ...listDL.map(i => `dl${i}.jpg`)
        ].sort(() => Math.random() - 0.5);
        
        allPhotos.forEach(src => {
            kyucContainer.innerHTML += `<div class="reveal rounded-2xl overflow-hidden shadow-2xl"><img src="${src}" class="w-full hover:opacity-80 transition cursor-zoom-in"></div>`;
        });

        // Truyền thông (Video & Ảnh)
        const videoGrid = document.getElementById('video-grid');
        listVD.forEach(i => {
            videoGrid.innerHTML += `
                <div class="glass-card p-4 rounded-3xl">
                    <video controls class="w-full rounded-2xl"><source src="vd${i}.mp4" type="video/mp4"></video>
                    <p class="mt-4 text-[10px] text-gray-500 text-center uppercase tracking-widest font-bold">Highlight Clip 0${i}</p>
                </div>`;
        });
        renderImgs(allPhotos.slice(0, 30), '', '', 'media-photo-grid');

        // --- 3. ĐIỀU HƯỚNG & NHẠC ---
        function showPage(pageId) {
            document.querySelectorAll('.sub-page').forEach(p => p.classList.remove('active-page'));
            document.getElementById(pageId).classList.add('active-page');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        const audio = document.getElementById('bgMusic');
        const musicIcon = document.getElementById('musicIcon');

            if (!isPlaying) {

            } else {
                audio.pause(); musicIcon.classList.remove('disk-rotate'); status.innerText = "Đang dừng";
            }
            isPlaying = !isPlaying;
        };
